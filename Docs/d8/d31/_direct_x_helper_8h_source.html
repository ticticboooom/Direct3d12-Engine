64 D3DENGINE_API inline HRESULT ReadDataFromFile(LPCWSTR filename, byte** data, UINT* size)
65 {
66  using namespace Microsoft::WRL;
67 
68  CREATEFILE2_EXTENDED_PARAMETERS extendedParams = {};
69  extendedParams.dwSize = sizeof(CREATEFILE2_EXTENDED_PARAMETERS);
70  extendedParams.dwFileAttributes = FILE_ATTRIBUTE_NORMAL;
71  extendedParams.dwFileFlags = FILE_FLAG_SEQUENTIAL_SCAN;
72  extendedParams.dwSecurityQosFlags = SECURITY_ANONYMOUS;
73  extendedParams.lpSecurityAttributes = nullptr;
74  extendedParams.hTemplateFile = nullptr;
75 
76  Wrappers::FileHandle file(CreateFile2(filename, GENERIC_READ, FILE_SHARE_READ, OPEN_EXISTING, &extendedParams));
77  if (file.Get() == INVALID_HANDLE_VALUE)
78  {
79  throw std::exception();
80  }
81 
82  FILE_STANDARD_INFO fileInfo = {};
83  if (!GetFileInformationByHandleEx(file.Get(), FileStandardInfo, &fileInfo, sizeof(fileInfo)))
84  {
85  throw std::exception();
86  }
87 
88  if (fileInfo.EndOfFile.HighPart != 0)
89  {
90  throw std::exception();
91  }
92 
93  *data = reinterpret_cast<byte*>(malloc(fileInfo.EndOfFile.LowPart));
94  *size = fileInfo.EndOfFile.LowPart;
95 
96  if (!ReadFile(file.Get(), *data, fileInfo.EndOfFile.LowPart, nullptr, nullptr))
97  {
98  throw std::exception();
99  }
100
