250  for (UINT n = 0; n < c_frameCount; n++)

252  m_renderTargets[n] = nullptr;
253  m_fenceValues[n] = m_fenceValues[m_currentFrame];
254  }
255 
256  UpdateRenderTargetSize();
257 
258  // The width and height of the swap chain must be based on the window's
259  // natively-oriented width and height. If the window is not in the native
260  // orientation, the dimensions must be reversed.
261 
262  m_d3dRenderTargetSize.x = m_outputSize.x;
263  m_d3dRenderTargetSize.y = m_outputSize.y;
264 
265  UINT backBufferWidth = lround(m_d3dRenderTargetSize.x);
266  UINT backBufferHeight = lround(m_d3dRenderTargetSize.y);
267 
268  if (m_swapChain != nullptr)
269  {
270  // If the swap chain already exists, resize it.
271  HRESULT hr = m_swapChain->ResizeBuffers(c_frameCount, backBufferWidth, backBufferHeight, m_backBufferFormat, 0);
272 
273  if (hr == DXGI_ERROR_DEVICE_REMOVED || hr == DXGI_ERROR_DEVICE_RESET)
274  {
275  // If the device was removed for any reason, a new device and swap chain will need to be created.
276  m_deviceRemoved = true;
277 
278  // Do not continue execution of this method. DeviceResources will be destroyed and re-created.
279  return;
280  }
281  else
282  {
283  ThrowIfFailed(hr);
284  }
285  }
286  else
287  {
288  // Otherwise, create a new one using the same adapter as the existing Direct3D device.
289  DXGI_SCALING scaling = DXGI_SCALING_NONE;
290  DXGI_SWAP_CHAIN_DESC1 swapChainDesc = {};
291 
292  swapChainDesc.Width = backBufferWidth; // Match the size of the window.
293  swapChainDesc.Height = backBufferHeight;
294  swapChainDesc.Format = m_backBufferFormat;
295  swapChainDesc.Stereo = false;
296  swapChainDesc.SampleDesc.Count = 1; // Don't use multi-sampling.
297  swapChainDesc.SampleDesc.Quality = 0;
298  swapChainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
299  swapChainDesc.BufferCount = c_frameCount; // Use triple-buffering to minimize latency.
300  swapChainDesc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD; // All Windows Universal apps must use _FLIP_ SwapEffects.
301  swapChainDesc.Flags = 0;
302  swapChainDesc.Scaling = scaling;
303  swapChainDesc.AlphaMode = DXGI_ALPHA_MODE_IGNORE;
304

306  ThrowIfFailed(
307  m_dxgiFactory->CreateSwapChainForHwnd(
308  m_commandQueue.Get(),
309  m_window,
310  &swapChainDesc,
311  nullptr,
312  nullptr,
313  &swapChain
314  )
315  );

362  dsvDesc.Flags = D3D12_DSV_FLAG_NONE;

394 void DX::DeviceResources::MoveToNextFrame()
395 {
396  const UINT64 currentFenceValue = m_fenceValues[m_currentFrame];
397  ThrowIfFailed(m_commandQueue->Signal(m_fence.Get(), currentFenceValue));
398 
399  // Advance the frame index.
400  m_currentFrame = m_swapChain->GetCurrentBackBufferIndex();
401 
402  // Check to see if the next frame is ready to start.
403  if (m_fence->GetCompletedValue() < m_fenceValues[m_currentFrame])
404  {
405  ThrowIfFailed(m_fence->SetEventOnCompletion(m_fenceValues[m_currentFrame], m_fenceEvent));
406  WaitForSingleObjectEx(m_fenceEvent, INFINITE, FALSE);
407  }
408 
409  // Set the fence value for the next frame.
410  m_fenceValues[m_currentFrame] = currentFenceValue + 1;
411 }
